# ES新特性速查指南

ECMAScript标准在不断演进，每个新版本都带来了实用的新特性。本文整理了ES7、ES8、ES9的主要特性，帮助快速了解和使用这些新功能。

# ES2016 (ES7) 新特性

# Array.prototype.includes()

用于判断数组是否包含某个元素，返回布尔值。

const arr = [1, 2, 3, NaN];

// 基本用法
arr.includes(2);       // true
arr.includes(4);       // false

// 与indexOf的区别
arr.indexOf(NaN);      // -1 (找不到)
arr.includes(NaN);     // true (能找到)

// 等价写法
arr.includes(2) === (arr.indexOf(2) !== -1)

# includes与indexOf的区别

  1. NaN处理：includes()能找到NaN，而indexOf()不行
  2. 返回值：includes()返回布尔值，indexOf()返回索引
  3. 可读性：includes()语义更明确

# 零值处理

includes()不会区分+0和-0：

[+0].includes(-0);  // true
[-0].includes(+0);  // true

# 指数运算符 (**)

提供了更简洁的幂运算语法：

// 新语法
2 ** 3;      // 8
2 ** 10;     // 1024

// 等价于
Math.pow(2, 3);   // 8
Math.pow(2, 10);  // 1024

// 支持赋值运算
let a = 2;
a **= 3;     // a = 8

# ES2017 (ES8) 新特性

# String Padding

为字符串添加填充字符，使其达到指定长度。

# padStart() 和 padEnd()

// padStart - 在开头填充
'es8'.padStart(2);          // 'es8' (已达到长度，不填充)
'es8'.padStart(5);          // '  es8' (默认用空格填充)
'es8'.padStart(6, 'woof');  // 'wooes8'
'es8'.padStart(14, 'wow');  // 'wowwowwowwoes8'
'es8'.padStart(7, '0');     // '0000es8'

// padEnd - 在末尾填充
'es8'.padEnd(2);          // 'es8'
'es8'.padEnd(5);          // 'es8  '
'es8'.padEnd(6, 'woof');  // 'es8woo'
'es8'.padEnd(14, 'wow');  // 'es8wowwowwowwo'
'es8'.padEnd(7, '6');     // 'es86666'

# 实际应用场景

时间格式化：

'8:00'.padStart(5, '0');   // '08:00'
'18:00'.padStart(5, '0');  // '18:00'

// 日期格式化
'12'.padStart(10, 'YYYY-MM-DD');       // "YYYY-MM-12"
'09-12'.padStart(10, 'YYYY-MM-DD');   // "YYYY-09-12"

# Object.values() 和 Object.entries()

对Object.keys()的补充，提供更丰富的对象遍历方式。

# Object.values()

获取对象所有可遍历属性的值：

const obj = { x: 'xxx', y: 1 };

Object.keys(obj);    // ['x', 'y']
Object.values(obj);  // ['xxx', 1]

// 数组（键为下标）
Object.values(['e', 's', '8']);  // ['e', 's', '8']

// 字符串（键为下标）
Object.values('es8');  // ['e', 's', '8']

// 纯数字键会按数字大小排序
Object.values({ 10: 'xxx', 1: 'yyy', 3: 'zzz' });  
// ['yyy', 'zzz', 'xxx']

# Object.entries()

获取对象所有可遍历属性的键值对：

const obj = { x: 'xxx', y: 1 };

Object.entries(obj);  // [['x', 'xxx'], ['y', 1]]

// 数组
Object.entries(['e', 's', '8']);  
// [['0', 'e'], ['1', 's'], ['2', '8']]

// 字符串
Object.entries('es8');  
// [['0', 'e'], ['1', 's'], ['2', '8']]

// 纯数字键按大小排序
Object.entries({ 10: 'xxx', 1: 'yyy', 3: 'zzz' });  
// [['1', 'yyy'], ['3', 'zzz'], ['10', 'xxx']]

# 对象遍历方式对比

for...in 循环：

const obj = { x: 'xxx', y: 1 };
for (let key in obj) {
    console.log(key);  // 'x', 'y'
}

for...of + Object.keys()：

for (let key of Object.keys(obj)) {
    console.log(key);  // 'x', 'y'
}

区别示例：

function Animal() {
    this.legs = 4;
}
function Human(name) {
    this.name = name;
}
Human.prototype = new Animal();
let human = new Human('es8');

// for...in 会遍历继承的属性
for (let key in human) {
    console.log(key);  // 'name', 'legs'
}

// Object.keys() 只遍历自身属性
for (let key of Object.keys(human)) {
    console.log(key);  // 'name'
}

# Object.getOwnPropertyDescriptors()

获取对象属性的描述符：

const obj = { es8: 'hello es8' };
Object.getOwnPropertyDescriptor(obj, 'es8');
// {
//   configurable: true,
//   enumerable: true,
//   value: "hello es8",
//   writable: true
// }

# 函数参数尾逗号

允许在函数定义和调用时使用尾逗号：

function es8(var1, var2, var3,) {
    // 尾逗号不会报错
}

es8(10, 20, 30,);  // 调用时也可以有尾逗号

这个特性方便了代码的版本控制，添加或删除参数时不会影响上一行的diff。

# Async Functions

Async/Await是ES8最重要的特性，从Generator和Promise演进而来：

// Promise方式
function fetchData() {
    return fetch('/api/data')
        .then(res => res.json())
        .then(data => console.log(data))
        .catch(err => console.error(err));
}

// Async/Await方式
async function fetchData() {
    try {
        const res = await fetch('/api/data');
        const data = await res.json();
        console.log(data);
    } catch (err) {
        console.error(err);
    }
}

# ES2018 (ES9) 新特性

# Rest/Spread 属性

在对象解构时使用剩余/扩展操作符：

// Rest - 收集剩余属性
const { a, b, ...rest } = { a: 1, b: 2, c: 3, d: 4 };
console.log(a);      // 1
console.log(b);      // 2
console.log(rest);   // { c: 3, d: 4 }

// Spread - 展开对象
const obj1 = { a: 1, b: 2 };
const obj2 = { ...obj1, c: 3 };  // { a: 1, b: 2, c: 3 }

// 实用技巧：删除对象属性（immutable方式）
const { password, ...userWithoutPassword } = user;

# 异步迭代

使用for await...of遍历异步迭代器：

async function process(array) {
    for await (let item of array) {
        console.log(item);
    }
}

// 异步生成器
async function* asyncGenerator() {
    yield await Promise.resolve(1);
    yield await Promise.resolve(2);
    yield await Promise.resolve(3);
}

(async () => {
    for await (let num of asyncGenerator()) {
        console.log(num);  // 1, 2, 3
    }
})();

# Promise.prototype.finally()

无论Promise成功还是失败，都会执行finally回调：

fetch('/api/data')
    .then(res => res.json())
    .then(data => {
        console.log(data);
    })
    .catch(err => {
        console.error(err);
    })
    .finally(() => {
        // 隐藏loading，无论成功还是失败
        hideLoading();
    });

# 使用场景

// 典型场景：API请求时的loading状态
async function fetchUserData() {
    showLoading();
    try {
        const data = await fetch('/api/user').then(res => res.json());
        updateUI(data);
    } catch (error) {
        showError(error);
    } finally {
        hideLoading();  // 无论成功失败都隐藏loading
    }
}

# 小结

ES标准的持续演进为JavaScript带来了更强大的功能和更优雅的语法。ES7的includes()和指数运算符提升了代码可读性；ES8的字符串填充、对象方法和Async/Await极大改善了开发体验；ES9的对象Rest/Spread、异步迭代和finally()则进一步完善了语言特性。

在实际项目中，可以通过Babel等工具将这些新特性转换为兼容旧浏览器的代码，从而在保证兼容性的同时享受新特性带来的便利。
