# 前端登录请求单例模式避免重复调用

在单页应用（SPA）中，当凭证失效时，多个页面或组件可能同时发起登录请求，导致重复调用登录接口。这里分享一个使用单例模式解决这个问题的方案。

# 问题场景

在以下情况下容易出现重复登录请求：

  1. 多个页面同时发起 API 请求
  2. Token 过期时多个请求同时发现需要重新登录
  3. 用户快速切换页面触发多次登录
  4. 并发请求在等待登录结果时继续触发新的登录

# 解决思路

使用单例模式：

  • 第一次登录请求时创建实例并缓存
  • 后续请求复用这个实例
  • 登录完成后清除实例，为下次登录做准备

# 实现方案

/**
 * 解决前端 首次登录/凭证失效重新登录 多页面等待异步请求返回，防止多次调用登录接口
 */
let { sleep, rand } = require('../utils/util')

// 当前登录请求单例
let loginReqIns

/**
 * 登录请求单例
 */
function loginReqMockIns() {
  if (loginReqIns) {
    console.log('使用已存在的登录实例')
    return loginReqIns
  } else {
    console.log('创建新的登录实例')
    loginReqIns = requestMock('login api')
    loginReqIns.then(() => {
      loginReqIns = null
      console.log(`清除登录实例`)
    })
    return loginReqIns
  }
}

/**
 * 模拟通用请求
 */
async function requestMock(api) {
  let delay = rand(100, 500)
  console.log(`请求 ${api} 开始 ${delay}ms`)
  await sleep(delay)
  console.log(`请求 ${api} 结束 ${delay}ms`)
  return true
}

// 测试场景：多个请求同时需要登录
;(async () => {
  console.log(`等待登录 1 开始`)
  let res = await loginReqMockIns()
  console.log(`等待登录 1 结束`)
})()

;(async () => {
  console.log(`等待登录 2 开始`)
  let res = await loginReqMockIns()
  console.log(`等待登录 2 结束`)
})()

;(async () => {
  await sleep(1000)
  console.log(`等待登录 3 开始`)
  let res = await loginReqMockIns()
  console.log(`等待登录 3 结束`)
})()

;(async () => {
  await sleep(1000)
  console.log(`等待登录 4 开始`)
  let res = await loginReqMockIns()
  console.log(`等待登录 4 结束`)
})()

# 实际应用

# 在 Axios 拦截器中使用

import axios from 'axios'

let loginPromise = null

// 响应拦截器
axios.interceptors.response.use(
  response => response,
  async error => {
    const { config, response } = error
    
    // 判断是否需要重新登录
    if (response?.status === 401) {
      // 如果已经有登录请求在进行，等待它完成
      if (loginPromise) {
        await loginPromise
        // 登录完成后重试原请求
        return axios.request(config)
      }
      
      // 创建新的登录请求
      loginPromise = login()
      
      try {
        await loginPromise
        // 登录成功后重试原请求
        return axios.request(config)
      } catch (err) {
        return Promise.reject(err)
      } finally {
        // 清除登录实例
        loginPromise = null
      }
    }
    
    return Promise.reject(error)
  }
)

async function login() {
  // 执行登录逻辑
  const response = await axios.post('/api/login', {
    // 登录凭证
  })
  
  // 保存新的 token
  localStorage.setItem('token', response.data.token)
  
  return response
}

# 在 Vue 应用中使用

// auth.js
class AuthManager {
  constructor() {
    this.loginPromise = null
  }
  
  async ensureLoggedIn() {
    if (this.loginPromise) {
      console.log('等待现有登录请求完成')
      return this.loginPromise
    }
    
    if (this.isTokenValid()) {
      return Promise.resolve()
    }
    
    console.log('发起新的登录请求')
    this.loginPromise = this.performLogin()
    
    try {
      await this.loginPromise
    } finally {
      this.loginPromise = null
    }
  }
  
  async performLogin() {
    // 执行实际的登录逻辑
    const response = await api.login()
    this.saveToken(response.token)
  }
  
  isTokenValid() {
    // 检查 token 是否有效
    const token = localStorage.getItem('token')
    return token && !this.isTokenExpired(token)
  }
  
  saveToken(token) {
    localStorage.setItem('token', token)
  }
}

export const authManager = new AuthManager()

// 在请求拦截器中使用
axios.interceptors.request.use(async config => {
  await authManager.ensureLoggedIn()
  config.headers.Authorization = `Bearer ${localStorage.getItem('token')}`
  return config
})

# 在 React 应用中使用

// useAuth.js
import { useRef, useCallback } from 'react'

export function useAuth() {
  const loginPromiseRef = useRef(null)
  
  const ensureLoggedIn = useCallback(async () => {
    if (loginPromiseRef.current) {
      console.log('等待现有登录请求')
      return loginPromiseRef.current
    }
    
    if (isTokenValid()) {
      return Promise.resolve()
    }
    
    console.log('创建新的登录请求')
    loginPromiseRef.current = performLogin()
    
    try {
      await loginPromiseRef.current
    } finally {
      loginPromiseRef.current = null
    }
  }, [])
  
  return { ensureLoggedIn }
}

# 方案优势

  1. 避免重复请求：多个并发请求只会触发一次登录
  2. 用户体验好：不会出现多次登录弹窗
  3. 性能优化：减少不必要的网络请求
  4. 代码简洁：使用单例模式，逻辑清晰

# 注意事项

  1. 内存泄漏：确保登录完成后清除实例
  2. 错误处理：登录失败时也要清除实例
  3. 超时处理：添加超时机制，避免永久等待
  4. 并发安全：JavaScript 单线程特性保证了这个方案的安全性

# 扩展优化

# 添加超时控制

function loginReqWithTimeout() {
  if (loginReqIns) {
    return loginReqIns
  }
  
  const timeoutPromise = new Promise((_, reject) => {
    setTimeout(() => reject(new Error('登录超时')), 30000)
  })
  
  const loginPromise = performLogin()
  
  loginReqIns = Promise.race([loginPromise, timeoutPromise])
    .finally(() => {
      loginReqIns = null
    })
  
  return loginReqIns
}

# 添加重试机制

async function loginReqWithRetry(maxRetries = 3) {
  if (loginReqIns) {
    return loginReqIns
  }
  
  loginReqIns = (async () => {
    for (let i = 0; i < maxRetries; i++) {
      try {
        return await performLogin()
      } catch (error) {
        if (i === maxRetries - 1) throw error
        console.log(`登录失败，重试 ${i + 1}/${maxRetries}`)
        await sleep(1000 * (i + 1))
      }
    }
  })()
  
  loginReqIns.finally(() => {
    loginReqIns = null
  })
  
  return loginReqIns
}

# 小结

使用单例模式管理登录请求是一个简单而有效的方案，可以很好地解决前端应用中重复登录请求的问题。这个模式不仅适用于登录场景，也可以应用到其他需要避免重复请求的场景中，如数据初始化、配置加载等。
