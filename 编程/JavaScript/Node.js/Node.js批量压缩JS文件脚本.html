# Node.js批量压缩JS文件脚本

在前端项目开发中，有时需要批量压缩 JavaScript 文件以减小体积、提升加载速度。这里分享一个使用 uglify-js 库递归压缩目录下所有 JS 文件的脚本。

# 使用场景

  • 小程序项目发布前的代码压缩
  • 旧项目迁移，需要压缩历史代码
  • 手动优化未经构建工具处理的项目
  • 批量处理第三方库文件

# 安装依赖

npm install uglify-js --save-dev

# 脚本实现

/**
 * JS文件压缩批处理
 */
const uglify = require('uglify-js');
const fs = require('fs');
const path = require('path');

// 指定要处理的目录
const filePath = `${__dirname}/weapp`;
// const filePath = `F:/wn/workspace/FXZ/tsg/tsg_wechat/weapp`;

console.log('开始处理');
fileDisplay(filePath);
console.log('处理结束');

/**
 * 递归遍历目录并压缩 JS 文件
 */
function fileDisplay(filePath) {
  let files = fs.readdirSync(filePath);

  files.forEach(function(filename) {
    let filedir = path.join(filePath, filename);
    let stats = fs.statSync(filedir);
    let isFile = stats.isFile(); // 是文件
    let isDir = stats.isDirectory(); // 是文件夹

    if (isFile) {
      // 只处理 .js 文件
      if (path.parse(filedir).ext == '.js') {
        let content = fs.readFileSync(filedir, 'utf-8');
        let min = uglify.minify(content);
        fs.writeFileSync(filedir, min.code);
        console.log(`压缩 ${filedir}`);
      }
    }

    if (isDir) {
      fileDisplay(filedir); // 递归处理子目录
    }
  });
}

# 代码说明

# 1. 目录遍历

function fileDisplay(filePath) {
  let files = fs.readdirSync(filePath);
  // 遍历文件和文件夹...
}

使用 fs.readdirSync 同步读取目录内容。

# 2. 文件类型判断

let stats = fs.statSync(filedir);
let isFile = stats.isFile();
let isDir = stats.isDirectory();

通过文件状态判断是文件还是目录。

# 3. JS 文件压缩

if (path.parse(filedir).ext == '.js') {
  let content = fs.readFileSync(filedir, 'utf-8');
  let min = uglify.minify(content);
  fs.writeFileSync(filedir, min.code);
  console.log(`压缩 ${filedir}`);
}
  • 读取 JS 文件内容
  • 使用 uglify.minify() 压缩
  • 将压缩后的代码写回原文件

# 4. 递归处理

if (isDir) {
  fileDisplay(filedir); // 递归
}

如果是目录，递归调用自身继续处理。

# 使用方法

  1. 安装依赖

    npm install uglify-js

  2. 保存脚本

    • 将代码保存为 compress-js.js

  3. 修改路径

    • 修改 filePath 为你的项目路径

  4. 执行脚本

    node compress-js.js

  5. 查看输出

    • 脚本会显示每个被压缩的文件路径

# 改进版本

# 1. 添加错误处理

function fileDisplay(filePath) {
  try {
    let files = fs.readdirSync(filePath);
    
    files.forEach(function(filename) {
      try {
        let filedir = path.join(filePath, filename);
        let stats = fs.statSync(filedir);
        
        if (stats.isFile() && path.parse(filedir).ext === '.js') {
          let content = fs.readFileSync(filedir, 'utf-8');
          let result = uglify.minify(content);
          
          if (result.error) {
            console.error(`压缩失败 ${filedir}:`, result.error);
            return;
          }
          
          fs.writeFileSync(filedir, result.code);
          console.log(`✓ 压缩成功 ${filedir}`);
        }
        
        if (stats.isDirectory()) {
          fileDisplay(filedir);
        }
      } catch (err) {
        console.error(`处理文件失败 ${filename}:`, err.message);
      }
    });
  } catch (err) {
    console.error(`读取目录失败 ${filePath}:`, err.message);
  }
}

# 2. 添加备份功能

function compressWithBackup(filedir) {
  const content = fs.readFileSync(filedir, 'utf-8');
  const result = uglify.minify(content);
  
  if (result.error) {
    console.error(`压缩失败 ${filedir}:`, result.error);
    return;
  }
  
  // 创建备份
  const backupPath = filedir + '.backup';
  fs.writeFileSync(backupPath, content);
  
  // 写入压缩后的代码
  fs.writeFileSync(filedir, result.code);
  console.log(`✓ 压缩成功 ${filedir} (已备份)`);
}

# 3. 添加统计信息

let stats = {
  total: 0,
  success: 0,
  failed: 0,
  originalSize: 0,
  compressedSize: 0
};

function fileDisplay(filePath) {
  let files = fs.readdirSync(filePath);
  
  files.forEach(function(filename) {
    let filedir = path.join(filePath, filename);
    let fileStats = fs.statSync(filedir);
    
    if (fileStats.isFile() && path.parse(filedir).ext === '.js') {
      stats.total++;
      const originalSize = fileStats.size;
      stats.originalSize += originalSize;
      
      let content = fs.readFileSync(filedir, 'utf-8');
      let result = uglify.minify(content);
      
      if (!result.error) {
        fs.writeFileSync(filedir, result.code);
        const compressedSize = fs.statSync(filedir).size;
        stats.compressedSize += compressedSize;
        stats.success++;
        
        const ratio = ((1 - compressedSize / originalSize) * 100).toFixed(2);
        console.log(`✓ ${filedir} (压缩率: ${ratio}%)`);
      } else {
        stats.failed++;
        console.error(`✗ ${filedir}:`, result.error);
      }
    }
    
    if (fileStats.isDirectory()) {
      fileDisplay(filedir);
    }
  });
}

// 执行完成后显示统计
fileDisplay(filePath);
console.log('\n压缩完成！');
console.log(`总文件数: ${stats.total}`);
console.log(`成功: ${stats.success}`);
console.log(`失败: ${stats.failed}`);
console.log(`原始大小: ${(stats.originalSize / 1024).toFixed(2)} KB`);
console.log(`压缩后: ${(stats.compressedSize / 1024).toFixed(2)} KB`);
console.log(`总压缩率: ${((1 - stats.compressedSize / stats.originalSize) * 100).toFixed(2)}%`);

# 注意事项

  1. 备份重要文件：压缩会直接覆盖原文件，建议先备份或使用版本控制
  2. 测试验证：压缩后务必测试功能是否正常
  3. 已压缩的文件：对已经压缩过的文件再次压缩效果不明显
  4. 语法错误：如果原文件有语法错误，压缩会失败
  5. 性能考虑：大型项目建议使用构建工具（如 webpack、vite）而不是这种批处理方式

# 小结

这个脚本适合小型项目或临时批量压缩的场景。对于正式的前端项目，建议使用专业的构建工具链。但了解这种批处理方式，在特殊情况下还是很有用的。
