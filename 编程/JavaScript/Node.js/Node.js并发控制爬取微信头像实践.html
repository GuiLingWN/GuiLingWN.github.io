# Node.js并发控制爬取微信头像实践

在实际项目中，有时需要批量下载用户头像等资源。如果不加控制地并发请求，很容易导致内存溢出或被服务器限流。这里分享一个带并发控制的批量下载脚本。

# 使用场景

  • 批量下载用户头像
  • 数据迁移时的资源同步
  • 离线备份图片资源
  • 批量下载网络资源

# 核心功能

  1. 并发控制：限制同时进行的下载任务数
  2. 数据库查询：从数据库获取待下载列表
  3. 批次处理：分组执行，便于监控进度
  4. 错误处理：单个任务失败不影响整体流程

# 脚本实现

/* 查询数据库并爬取微信用户头像 可控制并发 */
let http = require('https')
const fs = require('fs-extra')
const { query, init, close } = require('../../utils/db')

const dbName = 'database_name'

// 初始化数据库连接（已脱敏）
init({
  HOST: '数据库地址',
  USER: '用户名',
  PSWD: '密码',
  DB: dbName,
  PORT: 3306,
})

;(async () => {
  // 查询用户列表
  let sql = `select * from sys_user`
  let list = await query(sql)
  
  let pushAsyncList = []
  let folder = dbName
  await fs.ensureDir(`${__dirname}/${folder}/`)
  
  // 构建下载任务列表
  for (let index = 0; index < list.length; index++) {
    let { user_name, head_url } = list[index]
    let userName = fromBase64(user_name).replace(/[\\/:*?"<>|]/g, '')
    let url = head_url.replace('http://', 'https://')
    
    if (head_url) {
      pushAsyncList.push([downloadImg, { url, userName, folder }])
    }
  }
  
  try {
    await callBatchAsync(pushAsyncList, 100, (index) => {
      console.log(`第 ${index + 1} 组下载完成`)
    })
  } catch (error) {
    console.log(error)
  }
  
  close()
})()

/**
 * 批量异步调用，支持并发控制
 * @param {Array} fucList 异步函数列表 [[function, params], ...]
 * @param {Number} concurrent 并发量
 * @param {Function} groupEndCb 每组结束回调
 */
async function callBatchAsync(fucList, concurrent, groupEndCb) {
  let resList = []
  let fucListGroups = sliceArray(fucList, concurrent || 10)
  
  for (let i = 0; i < fucListGroups.length; i++) {
    let list = []
    fucListGroups[i].forEach((cbInfo) => {
      list.push(cbInfo[0](cbInfo[1]))
    })
    
    try {
      let res = await Promise.all(list)
      resList.push(...res)
      groupEndCb && groupEndCb(i, res)
    } catch (error) {
      console.log(error)
    }
  }
  
  return resList
}

/**
 * 将一维数组分割成二维数组
 * @param {Array} array 原数组
 * @param {Number} size 每组大小
 */
function sliceArray(array, size) {
  let result = []
  for (let x = 0; x < Math.ceil(array.length / size); x++) {
    let start = x * size
    let end = start + size
    result.push(array.slice(start, end))
  }
  return result
}

/**
 * Base64 解码
 */
function fromBase64(content) {
  return Buffer.from(content, 'base64').toString()
}

/**
 * 下载图片
 */
async function downloadImg({ url, userName, folder = 'img' }) {
  return new Promise((resolve, reject) => {
    http
      .get(url, function (res) {
        let imgData = ''
        res.setEncoding('binary') // 设置为 binary 编码
        
        res.on('data', function (chunk) {
          imgData += chunk
        })
        
        res.on('end', function () {
          fs.writeFile(
            `${__dirname}/${folder}/${userName}.jpg`,
            imgData,
            'binary',
            function (err) {
              if (err) {
                console.log(`下载失败 ${userName} ${url}`)
                reject()
                return
              }
              resolve()
            }
          )
        })
      })
      .on('error', (e) => {
        console.error(e)
        reject()
      })
  })
}

工作原理：

  1. 将任务列表按并发数分组
  2. 每组使用 Promise.all 并发执行
  3. 等待一组完成后再执行下一组
  4. 通过回调函数监控进度

# 2. 数组分组

将大数组切分成指定大小的小数组。

# 改进建议

# 1. 添加重试机制

async function downloadWithRetry(params, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await downloadImg(params)
    } catch (error) {
      if (i === maxRetries - 1) throw error
      console.log(`重试 ${params.userName} (${i + 1}/${maxRetries})`)
      await sleep(1000 * (i + 1))
    }
  }
}

# 3. 使用更现代的下载方式

const axios = require('axios')

async function downloadImg({ url, userName, folder = 'img' }) {
  const response = await axios.get(url, {
    responseType: 'arraybuffer'
  })
  
  await fs.writeFile(
    `${__dirname}/${folder}/${userName}.jpg`,
    response.data
  )
}

# 注意事项

  1. 并发数控制：根据服务器性能和网络带宽调整，避免过大导致问题
  2. 错误处理：单个任务失败不应影响整体流程
  3. 内存管理：大量下载时注意内存占用
  4. 文件名安全：过滤特殊字符，避免文件系统错误
  5. 敏感信息：生产环境中不要硬编码数据库密码

# 小结

这个脚本展示了如何在 Node.js 中实现带并发控制的批量下载。核心思想是将任务分组，利用 Promise.all 并发执行，通过控制批次来限制同时进行的任务数。这种模式可以应用到各种需要并发控制的场景中。
