# Docker常用服务快速部署指南

使用Docker可以快速部署各种常用服务，本文整理了一些常见服务的Docker部署命令，包括数据库、开发工具等。

# 数据库服务

# MySQL 8.0

# 拉取镜像
docker pull mysql:8.0.21

# 启动容器（注意设置表名不区分大小写）
docker run -d --name mysql -p 3306:3306 \
  -e MYSQL_ROOT_PASSWORD=your_password \
  mysql:8.0.21 --lower_case_table_names=1

# 进入容器修改密码
docker exec -it mysql bash
mysql -uroot -p
use mysql;
ALTER USER 'root'@'%' IDENTIFIED WITH mysql_native_password BY '新密码';
flush privileges;

# MySQL 5.7

docker pull mysql:5.7.39
docker run --restart=always -itd --name mysql \
  -p 33060:3306 \
  -e MYSQL_ROOT_PASSWORD=your_password \
  mysql:5.7.39

# Redis

# 拉取镜像
docker pull redis:latest

# 启动容器（设置密码并允许外部访问）
docker run --restart=always -d --name redis \
  -p 16379:6379 \
  redis --requirepass "your_password" --bind "0.0.0.0"

# 关闭保护模式的启动方式
docker run --restart=always -d --name redis \
  -p 6379:6379 \
  redis --requirepass "your_password" --bind "0.0.0.0" --protected-mode no

# MongoDB

# 拉取镜像
docker pull mongo:5.0.9

# 启动容器（启用认证）
docker run -itd --name mongo -p 27017:27017 mongo:5.0.9 --auth

# 创建管理员用户
docker exec -it mongo mongo admin
> db.createUser({ 
    user:'admin',
    pwd:'your_password',
    roles:[
      { role:'userAdminAnyDatabase', db: 'admin'},
      "readWriteAnyDatabase"
    ]
  });

# 验证用户登录
> db.auth('admin', 'your_password')

# Web服务

# Nginx

使用host网络模式部署Nginx，监听所有端口：

docker run \
  -v /opt/docker-nginx/nginx.conf:/etc/nginx/nginx.conf \
  -v /opt/docker-nginx/logs:/var/log/nginx \
  -v /opt/docker-nginx/html:/usr/share/nginx/html \
  -v /opt/docker-nginx/conf:/etc/nginx/conf.d \
  -v /etc/localtime:/etc/localtime \
  --net host \
  --name nginx \
  --restart=always \
  -d nginx

# Nginx代理转发

用于代理NAS、WireGuard等服务：

docker run -d \
  --name nginx-proxy \
  --restart=always \
  -p 12580:5000 \
  -p 12581:51820/udp \
  -v /data/nginx-proxy/nginx.conf:/etc/nginx/nginx.conf \
  -v /data/nginx-proxy/conf.d:/etc/nginx/conf.d \
  -v /data/nginx-proxy/logs:/var/log/nginx \
  -v /etc/localtime:/etc/localtime \
  nginx:alpine

nginx.conf中的stream配置示例（与http配置同级）：

stream {
    log_format basic '$remote_addr [$time_local] $protocol $status $bytes_sent $bytes_received $session_time';
    access_log /var/log/nginx/stream_access.log basic;

    server {
        listen 5000;
        proxy_pass 192.168.2.22:5000;
        proxy_connect_timeout 5s;
        proxy_timeout 1m;
    }

    server {
        listen 51820 udp;
        proxy_pass 192.168.2.8:51820;
        proxy_connect_timeout 5s;
        proxy_timeout 1m;
    }
}

# 开发工具

# YAPI接口管理平台

# 拉取镜像
docker pull registry.cn-hangzhou.aliyuncs.com/anoyi/yapi

# 配置文件（/data/yapi/config.json）
{
  "port": "3000",
  "adminAccount": "admin@anoyi.com",
  "timeout":120000,
  "db": {
    "servername": "mongo",
    "DATABASE": "yapi",
    "port": 27017,
    "user": "anoyi",
    "pass": "anoyi.com",
    "authSource": "admin"
  }
}

# 初始化数据库
docker run -it --rm \
  --link mongo:mongo \
  --entrypoint npm \
  --workdir /yapi/vendors \
  -v /data/yapi/config.json:/yapi/config.json \
  registry.cn-hangzhou.aliyuncs.com/anoyi/yapi \
  run install-server

# 启动服务
docker run -d \
  --name yapi \
  --link mongo:mongo \
  --workdir /yapi/vendors \
  -p 3000:3000 \
  -v /data/yapi/config.json:/yapi/config.json \
  registry.cn-hangzhou.aliyuncs.com/anoyi/yapi \
  server/app.js

# 音乐服务Navidrome

Navidrome是一个开源的音乐流媒体服务器。

# 单独安装

docker run -d \
   --name navidrome \
   --restart=unless-stopped \
   -v /volume2/docker/navidrome/music:/music \
   -v /volume2/docker/navidrome/data:/data \
   -p 4533:4533 \
   -e ND_LOGLEVEL=info \
   deluan/navidrome:latest

# 完整配置（docker-compose）

version: '1'
services:
  navidrome:
    image: deluan/navidrome:latest
    ports:
      - '4533:4533'
    environment:
      ND_SCANSCHEDULE: 1m
      ND_SPOTIFY_ID: your_spotify_id
      ND_SPOTIFY_SECRET: your_spotify_secret
      ND_LASTFM_LANGUAGE: zh
      ND_LOGLEVEL: info
      ND_SESSIONTIMEOUT: 24h
      ND_ENABLETRANSCODINGCONFIG: 'true'
      ND_TRANSCODINGCACHESIZE: '4000M'
      ND_IMAGECACHESIZE: '1000M'
    volumes:
      - '/volume2/docker/navidrome/data:/data'
      - '/volume2/docker/navidrome/music:/music:ro'
    restart: always
  miniserve:
    image: svenstaro/miniserve:latest
    depends_on:
      - navidrome
    ports:
      - '4534:8080'
    volumes:
      - '/volume2/docker/navidrome/music:/downloads'
    command: '-r -z -u -q -p 8080 -a username:password /downloads'
    restart: always

# 注意事项

  1. 所有密码请根据实际情况修改，不要使用示例中的弱密码
  2. 挂载路径需根据实际服务器目录结构调整
  3. 端口映射请避免与现有服务冲突
  4. 建议为重要数据配置数据卷持久化
  5. 生产环境建议配置容器资源限制
