# Nginx 配置详解与常见问题解决

Nginx 是一个高性能的 HTTP 和反向代理服务器，也是一个 IMAP/POP3/SMTP 代理服务器。本文将详细介绍 Nginx 的配置要点以及常见问题的解决方案。

# Nginx 核心配置

# 日志格式配置

Nginx 提供了灵活的日志格式配置选项，可以根据需要自定义日志内容：

log_format main '$remote_addr - [$time_local] $host - "$request" '
                '$status $upstream_response_time $request_time "$http_referer"'
                '"$http_user_agent" "$http_x_forwarded_for" $body_bytes_sent ';
access_log /var/log/nginx/access.log main;
error_log /var/log/nginx/error.log;

# 超时和大小限制配置

# 限制请求体大小，超过设定值返回413错误
client_max_body_size 20m;

# 读取请求头的超时时间，超过设定值返回408错误
client_header_timeout 60s;

# 读取请求实体的超时时间，超过设定值返回413错误
client_body_timeout 60s;

# HTTP请求无法立即被后端处理时的等待时间
proxy_connect_timeout 60s;

# 后端处理完成后Nginx等待响应的时间
proxy_read_timeout 60s;

# 数据从后端返回给Nginx的超时时间
proxy_send_timeout 60s;

配置层级示例：

http {
  # 控制全局nginx所有请求报文大小
  # client_max_body_size 20m;
  
  server {
    # 控制该server的所有请求报文大小
    # client_max_body_size 20m;
    
    location a {
    }
    
    location b {
      # 控制满足该路由规则的请求报文大小
      # client_max_body_size 20m;
    }
  }
}

# 常见问题及解决方案

# 1. 域名与 server_name 不匹配也能访问

问题原因： 当 Nginx 匹配不到 server_name 时，会将请求交给默认 server 处理。如果没有配置 default_server 指令，则会交给第一个开启此端口的 server 处理。

解决方案：

  1. 新增一个默认 server 或将一个废弃 server 配置为 default_server
  2. 对于 Nginx 1.19.4 及以上版本，可以直接配置：
server {
    listen 80 default_server;
    listen 443 ssl default_server;
    ssl_reject_handshake on;
    server_name _;
}

# 2. 系统维护提示配置

在系统维护期间，可以通过以下配置向用户返回友好的提示信息：

server {
    error_page 502 503 =200 /maintainInfo;
    
    location = /maintainInfo {
        add_header 'Access-Control-Allow-Origin' '*' always;
        add_header 'Access-Control-Allow-Headers' '*' always;
        add_header 'Access-Control-Allow-Methods' '*' always;
        default_type application/json;
        return 200 '{"code":503,"message":"系统升级中，请稍后再试","data":null}';
    }
}

注意：不加 =200 时，返回的是原先的 HTTP 错误码；配上后如果出现 500 等错误都返回给用户 200 状态。

# 3. 调试技巧

在调试 Nginx 配置时，可以通过以下方式输出变量信息：

  1. 在 log_format 中打印变量
  2. 在 location 中 return 变量（设置响应为 txt/json）

示例：

http {
    map $host $subdomain {
        ~(.*)\.saas\.test\.cn $1;
    }
    
    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent" "$http_x_forwarded_for"'
                    'subdomain=$subdomain $host';

    server {
        listen 80;
        server_name *.saas.test.cn;
        
        location / {
            default_type application/json;
            return 200 '1$host 2$subdomain';
        }
    }
}

# SSL 证书配置

# 使用 acme.sh 生成免费证书

# 安装 acme.sh

curl https://get.acme.sh | sh

# 腾讯云自动申请证书

export DP_Id="DP_Id" 
export DP_Key="DP_Key" 
acme.sh --issue --dns dns_dp -d api.your_domain.com --force

# 华为云自动申请证书

export HUAWEICLOUD_Username="HUAWEICLOUD_Username" # 用户名(统一身份认证服务)
export HUAWEICLOUD_Password="HUAWEICLOUD_Password" # 密码
export HUAWEICLOUD_DomainName="HUAWEICLOUD_DomainName" # 账号名

acme.sh --issue --dns dns_huaweicloud -d api.your_domain.com --force

# 使用 mkcert 为本地服务生成 HTTPS 证书

安装 mkcert：

sudo apt install libnss3-tools
curl -JLO "https://github.com/FiloSottile/mkcert/releases/latest/download/mkcert-v1.4.4-linux-amd64"
chmod +x mkcert-v1.4.4-linux-amd64
sudo mv mkcert-v1.4.4-linux-amd64 /usr/local/bin/mkcert

生成证书：

# 安装本地根证书（仅需一次）
mkcert -install

# 为本地 IP 生成证书
mkcert 192.168.2.8

# 同时为多个域名/IP 生成
mkcert 192.168.2.8 localhost 127.0.0.1

Nginx SSL 配置示例：

server {
    listen 6753 ssl;
    server_name 192.168.2.8;

    ssl_certificate conf.d/ssl/192.168.2.8+2.pem;
    ssl_certificate_key conf.d/ssl/192.168.2.8+2-key.pem;

    ssl_session_cache shared:SSL:1m;
    ssl_session_timeout 5m;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;

    location / {
        proxy_pass http://localhost:6752;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        
        # WebSocket 配置
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
}

# 总结

Nginx 的配置虽然看起来复杂，但掌握了核心概念和常见问题的解决方案后，就能很好地应对各种场景。在实际使用中，建议：

  1. 合理配置日志格式，便于问题排查
  2. 根据业务需求设置合适的超时和大小限制
  3. 正确处理域名匹配和默认服务器配置
  4. 在系统维护时提供友好的用户提示
  5. 使用合适的工具管理 SSL 证书

通过不断实践和总结，可以更好地发挥 Nginx 的强大功能。
